\INSRChapter{Session Safety}

\ClinicalImplication{Use clinical judgement before any exposure step.}

\begin{ClinicalNote}
Confirm informed consent and current risk status before processing-oriented interventions.
\end{ClinicalNote}

\begin{SafetyBox}
Check signal quality before interpreting HRV or EDA data. Apply a somatic hold and return to stabilization when clinically indicated.
\end{SafetyBox}
